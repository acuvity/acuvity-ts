# Extraction

Represents the extracted information to log.

## Example Usage

```typescript
import { Extraction } from "@acuvity/acuvity/models/components";

let value: Extraction = {
  piIs: {
    "ssn": 0.8,
  },
  categories: [
    {
      group: "image",
      type: "png",
    },
  ],
  confidentiality: 0.9,
  customDataTypes: {
    "my_cdt": 1,
  },
  exploits: {
    "prompt_injection": 0.8,
  },
  intent: {
    "write": 0.8,
  },
  keywords: {
    "my_keywork": 0.8,
  },
  languages: {
    "english": 0.8,
  },
  malcontents: {
    "toxic": 0.8,
  },
  modalities: [
    {
      group: "image",
      type: "png",
    },
  ],
  relevance: 0.9,
  secrets: {
    "credentials": 0.7,
  },
  topics: {
    "category/enterprise": 0.7,
    "department/logistics": 0.8,
    "depict/document": 0.8,
    "extracted/typed_text_content": 1,
    "timeframe/current_year": 0.6,
  },
};
```
